// Naive Solution 
Time complexity = O(n)
Aux Space = O(1)

int distinct(vector<int>& arr) {
    int count = 0;
    for(int i = 0; i < arr.size(); i++) {
        bool flag = false;
        for(int j = 0; j < i; j++) {
            if(arr[i] == arr[j]) {
                flag = true;
                break;
            }
        }
        if(flag == false) {
            count++;
        }
    }
    return count;
    
}


// Efficient Solution
Time complexity = Theta(n)
Aux Space = O(n)
// insert function only insert if item is not present if it already present then it doesn't insert it.

int distinct(vector<int>& arr) {
unordered_set<int> s;
for(int i = 0; i < arr.size(); i++) {
    s.insert(arr[i]);
}
 return s.size();
    
}

Time complexity = Theta(n)
Aux space = O(n)
