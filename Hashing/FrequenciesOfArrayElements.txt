// Naive Solution 
// Time = O(n * n)
// Aux Space = O(1)

void printFreq(int arr[], int n) {
  for(int i = 0; i < n; i++) {
    bool flag = false;
    for(int j = 0; j < i; j++) {
      if(arr[i] == arr[j]) {
        flag = true;
        break;
      }
      
    }
    if(flag == true) continue;
    int freq = 1;
    for(int l = i+1; l < n; l++) {
      if(arr[i] == arr[l]) {
        freq++;
      }
    }
    cout<<arr[i]<<" "<<freq<<endl;
  }
}
