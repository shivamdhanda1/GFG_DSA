// Efficient Solution
// Time = O(n)

class Solution {
  public:
   
   
    bool subArrayExists(vector<int>& arr) {
    unordered_set<int> s; 
    int pre_sum = 0; 
    for(int i = 0; i < arr.size(); i++) {
        pre_sum += arr[i];
        if(s.find(pre_sum) != s.end()) {
            return true;
        }
        if(pre_sum == 0) {
            return true;
        }
            s.insert(pre_sum);
        
    }
    return false;


    }
};
