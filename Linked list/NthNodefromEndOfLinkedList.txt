// Naive approach 
// here we need to find the length of the linked list
class Solution {
  public:
    int getKthFromLast(Node* head, int k) {
        if(head == NULL) return NULL;
        int count = 0;
        Node *curr = head;
        
        while(curr != NULL) {
            count++;
            curr = curr->next;
        }
        
        curr = head;
        
        if(k > count) {
            return -1;
        }
        
        int m = count - k + 1;
        for(int i = 1; i < m; i++) {
            curr = curr->next;
    }
    
    return curr->data;
    }
};
