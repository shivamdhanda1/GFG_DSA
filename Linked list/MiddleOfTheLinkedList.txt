// Naive approach 
// i.e this requries two traversals 
// Efficient approach is written below naive approach

class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        if(head == NULL) return NULL;
        
        ListNode *curr = head;
        int length = 1;
        while(curr->next != NULL) {
            curr = curr->next;
            length++;
        }
        curr = head;
        int mid = length/2;
        for(int i = 0; i < mid; i++) {
            curr = curr->next;
        }
        return curr;
    }
};

// Efficient Solution 
// It requires only 1 traversal
// Based on idea of slow and fast pointers i.e we maintain two pointers or references
// slow moves 1 node at a time and fast moves two node at a time so when fast reaches the end , slow reaches the middle // here we have to handle both even and odd linked list 
// here fast != NULL makes sure that NULL->next is never accessed , so preventing segmentation fault.

void printMiddle(Node *head) {
    if(head == NULL ) return;
    Node *slow = head, *fast = head;
    while(fast != NULL && fast->next != NULL) {
        slow = slow->next;
        fast = fast->next->next;
    }
    // when fast becomes NULL or fast next becomes NULL , then slow is referring to the middle node
    cout<< (slow->data);
}
