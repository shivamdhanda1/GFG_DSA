Node *insertEnd(Node *head, int data) {
  Node *temp = new Node(data);
  
  if(head == NULL) {
    return temp;
  }
  Node *curr = head;
  while(curr->next != NULL) { // curr->next is imp. condition because if we only wrote curr then curr will come out of loop with pointing to null
    curr = curr->next;
  }
  curr->next = temp;
  temp->prev = curr;
  return head;
}
